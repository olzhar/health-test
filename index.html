        // Îòîáðàæåíèå ðåçóëüòàòîâ

        function displayResults(scores) {

            const resultsBody = document.getElementById('resultsBody');

            resultsBody.innerHTML = '';

           

            scores.forEach((score, index) => {

                const evaluation = getEvaluation(score, index);

                const row = document.createElement('tr');

                row.innerHTML = `

                    <td>${systemNames[index]}</td>

                    <td><strong>${score}</strong></td>

                    <td class="status-${evaluation.class}">${evaluation.text}</td>

                `;

                resultsBody.appendChild(row);

            });

           

            document.getElementById('results').style.display = 'block';

            document.getElementById('results').scrollIntoView({ behavior: 'smooth' });

        }

       

        // Îïðåäåëåíèå îöåíêè

        function getEvaluation(score, systemIndex) {

            const criteria = evaluationCriteria[systemIndex];

            

            if (score >= criteria.veryGood[0] && score <= criteria.veryGood[1]) {

                return { text: 'Î÷åíü õîðîøî', class: 'very-good' };

            } else if (score >= criteria.good[0] && score <= criteria.good[1]) {

                return { text: 'Õîðîøî', class: 'good' };

            } else if (score >= criteria.satisfactory[0] && score <= criteria.satisfactory[1]) {

                return { text: 'Óäîâëåòâîðèòåëüíî', class: 'satisfactory' };

            } else {

                return { text: 'Íåóäîâëåòâîðèòåëüíî', class: 'unsatisfactory' };

            }

        }

       

        // Îòïðàâêà ðåçóëüòàòîâ ÷åðåç EmailJS

        async function sendResultsToEmail(scores) {

            const personalInfo = {

                surname: document.getElementById('surname').value,

                name: document.getElementById('name').value,

                patronymic: document.getElementById('patronymic').value,

                year: document.getElementById('year').value,

                phone: document.getElementById('phone').value

            };

           

            const results = scores.map((score, index) => ({

                system: systemNames[index],

                score: score,

                evaluation: getEvaluation(score, index).text

            }));

           

            // Ôîðìèðîâàíèå òåêñòà ðåçóëüòàòîâ äëÿ email

            let resultsText = '';

            let problemSystems = [];

            let satisfactorySystems = [];

            let goodSystems = [];

           

            results.forEach(result => {

                resultsText += `${result.system}: ${result.score} áàëëîâ (${result.evaluation})\n`;

               

                if (result.evaluation === 'Íåóäîâëåòâîðèòåëüíî') {

                    problemSystems.push(result.system);

                } else if (result.evaluation === 'Óäîâëåòâîðèòåëüíî') {

                    satisfactorySystems.push(result.system);

                } else {

                    goodSystems.push(result.system);

                }

            });

           

            // Àíàëèç ðåçóëüòàòîâ

            let analysis = '\n=== ÀÍÀËÈÇ ÐÅÇÓËÜÒÀÒÎÂ ===\n';

            if (problemSystems.length > 0) {

                analysis += `⚠️ ÒÐÅÁÓÞÒ ÂÍÈÌÀÍÈß (${problemSystems.length} ñèñòåì):\n${problemSystems.map(s => '- ' + s).join('\n')}\n\n`;

            }

            if (satisfactorySystems.length > 0) {

                analysis += `⚡ Óäîâëåòâîðèòåëüíîå ñîñòîÿíèå (${satisfactorySystems.length} ñèñòåì):\n${satisfactorySystems.map(s => '- ' + s).join('\n')}\n\n`;

            }

            if (goodSystems.length > 0) {

                analysis += `✅ Õîðîøåå/Îòëè÷íîå ñîñòîÿíèå (${goodSystems.length} ñèñòåì):\n${goodSystems.map(s => '- ' + s).join('\n')}\n\n`;

            }

           

            // Ïàðàìåòðû äëÿ EmailJS øàáëîíà

            const templateParams = {

                surname: personalInfo.surname,

                name: personalInfo.name,

                patronymic: personalInfo.patronymic || 'íå óêàçàíî',

                year: personalInfo.year,

                phone: personalInfo.phone,

                date: new Date().toLocaleDateString('ru-RU'),

                time: new Date().toLocaleTimeString('ru-RU'),

                results: resultsText,

                analysis: analysis,

                total_problems: problemSystems.length,

                total_satisfactory: satisfactorySystems.length,

                total_good: goodSystems.length

            };

           

            // Îòïðàâêà ÷åðåç EmailJS

            return emailjs.send(EMAILJS_SERVICE_ID, EMAILJS_TEMPLATE_ID, templateParams);

        }
